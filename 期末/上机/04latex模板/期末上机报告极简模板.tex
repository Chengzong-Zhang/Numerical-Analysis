\section{梯形方法、BDF2、改进Euler方法比较}

\subsection{问题描述}

分别利用梯形方法、BDF2格式数值求解如下问题
\[
  y'=-exp(10t)(y-sin(t))+cos(t),\qquad
  0 <t\le 1,\qquad y(0)=0
\]


\begin{enumerate}
  \item 分别取步长$h=0.1,0.05,0.025,0.0125,0.00625$进行计算，根据数值结果计算两种方法的误差收敛阶，并对计算结果进行比较分析。
  \item 如利用改进的 Euler 方法会出现什么样的结果，试对计算结果进行分析。
\end{enumerate}

\subsection{方法介绍}

梯形方法对积分使用了更精确的梯形公式，是隐式离散化方法：
\[
  y_{n+1}=y_n+\frac{h}{2}(f_n+f_{n+1}),
\]

BDF方法，即基于插值型数值微分构造的多步法，用 $(t_n,y_n),(t_{n-1},y_{n-1}),\cdots,(t_{n-k},y_{n-k})$ 共 $k+1$ 点构造插值多项式 $q_k(t)$，用 $q_k'(t_n)$ 近似 $f(t_n, y(t_n))$：

$$\sum_{m=1}^{k} \frac{\nabla^m y_n}{m} = hf(t_n, y_n)$$

通用形式：$y_n + \sum_{j=1}^{k} \alpha_j y_{n-j} = h\beta_0 f(t_n, y_n)$,
\\2阶公式为$y_n - \frac{4}{3}y_{n-1} + \frac{1}{3}y_{n-2} = \frac{2}{3}hf_n$
BDF方法对刚性方程有优良稳定性。BDF2 的绝对稳定区间比显式方法大得多

改进 Euler属于二阶显式 Runge--Kutta 方法。
\[
  K_1=f(x_n,y_n),\qquad K_2=f(x_n+h,y_n+hK_1),\qquad
  y_{n+1}=y_n+\frac{h}{2}(K_1+K_2).
\]

\subsection{结果分析}

\begin{enumerate}
  \item 梯形法 (Trapezoid): A-稳定，二阶精度。对于刚性方程 e^(10t)，表现稳健，误差随 h² 衰减。这表明梯形方法在该测试问题中充分发挥了其二阶精度，且数值稳定性良好，进入了渐近收敛
区。
  \item BDF2:A-稳定，二阶精度。起步值需 RK4 提供，牛顿迭代求解隐式方程，计算量略大但精度相当。收敛阶为负数且绝对值增大（如 `-22.3`），说明误差随步长减小而指数级增长，验证了其在特定条件下（可能是刚性方
程的大步长或小步长耦合）的不稳定性。
观察：若梯形法收敛阶接近 2.0，BDF2 也接近 2.0，则说明两者在该问题下均发挥预期性能。
  \item 改进 Euler:刚性特征：由于 ODE 中包含项 -exp(10t)y，特征值 λ(t) ≈ -exp(10t)，在 t=1 时 λ ≈ -22026。Improved Euler (显式) 的绝对稳定区域约为 h*λ > -4.5，即 h < 0.0002 * exp(-10t)。在 t=1 附近，Improved Euler 需要极小的步长才能维持稳定，否则误差会随阶数平方指数增长。收敛阶极高（$>40$），这通常意味着在初始阶段误差对 $h$ 极度敏感，或者初始误差主要由舍入误差主导，导致
计算出的“阶”虚高；随着 $h$ 进一步减小，数值解可能进入精度极限。
\end{enumerate}

\subsection*{A.1\quad 题目一 Python 程序}

\begin{lstlisting}[style=python,caption={梯形方法、BDF2、改进Euler方法比较}]
import numpy as np

# ==================== 1. 定义基础函数 (全局作用域) ====================
def f(t, y):
    """ODE 右端项：y' = -exp(10t)*(y-sin(t)) + cos(t)"""
    return -np.exp(10 * t) * (y - np.sin(t)) + np.cos(t)

def fy(t, y):
    """偏导数 ∂f/∂y：用于牛顿迭代和隐式方程求解"""
    return -np.exp(10 * t)

def y_exact(t):
    """精确解函数（解析推导结果）"""
    # 根据 ODE 解析解推导，特解为 sin(t)，初值 y(0)=0 满足 sin(0)=0
    return np.sin(t)

# ==================== 2. 数值求解器 Solve_Step(h, method) ====================
def Solve_Step(h, method):
    # 全局参数 (假设与主程序一致)
    t0 = 0.0
    T = 1.0
    y0_val = 0.0

    # 计算总步数与时间网格
    N = int(round((T - t0) / h))
    t = t0 + np.arange(N + 1) * h

    # 预分配数值解数组
    y = np.zeros(N + 1, dtype=float)
    y[0] = y0_val

    # 迭代参数 (常量定义)
    tol, max_iter = 1e-6, 2000

    if method == "trapezoid":
        for n in range(N):
            tn, yn = t[n], y[n]
            z = yn + h * f(tn, yn)
            # Newton 迭代求解隐式方程：z - yn - (h/2)[f(tn,yn) + f(tn+h,z)] = 0
            for _ in range(max_iter):
                z_new = yn + h*(f(tn, yn)+f(t[n+1], z))/2  # 梯形公式 z=y_n+h/2*(f_n+f_{n+1})。
                if abs(z_new-z) < tol: break  # 达到迭代容差后停止。
                z = z_new  # 更新隐式未知量。
            y[n + 1] = z


    elif method == "BDF2":
        k = 2
        # BDF2 系数：标准形式 y_{n+1} - 4/3*y_n + 1/3*y_{n-1} = h * 2/3 * f(t_{n+1})
        # 对应系数 alpha (y_{n+1}, y_n, y_{n-1}): [1, -4/3, 1/3]
        # beta: 2/3
        coeffs_alpha = np.array([1.0, -4.0 / 3.0, 1.0 / 3.0])
        beta_val = 2.0 / 3.0

        # 起步阶段 (RK4 提供前 k-1=1 步值，即 y[0], y[1])
        for n_start in range(k - 1):
            tn = t[n_start]
            yn = y[n_start]
            
            q1 = f(tn, yn)
            q2 = f(tn + h / 2.0, yn + h * q1 / 2.0)
            q3 = f(tn + h / 2.0, yn + h * q2 / 2.0)
            q4 = f(tn + h, yn + h * q3)
            
            y[n_start + 1] = yn + h * (q1 + 2*q2 + 2*q3 + q4) / 6.0

        # BDF 主推进阶段
        for n in range(k - 1, N):
            # 计算已知项 (历史值移项到右侧，注意系数对应 y[n], y[n-1])
            known = coeffs_alpha[1] * y[n] + coeffs_alpha[2] * y[n - 1]
            
            z = y[n]  # 初始猜测
            
            # Newton 迭代求解隐式方程：z - known - h*beta*f(t_{n+1}, z) = 0
            for iteration in range(max_iter):
                G = z - known - h * beta_val * f(t[n + 1], z)
                
                dG = 1.0 - h * beta_val * fy(t[n + 1], z)
                
                if abs(dG) < 1e-15:
                    break
                
                delta = G / dG
                z_new = z - delta
                
                if abs(z_new - z) < tol:
                    z = z_new
                    break
                z = z_new
            
            y[n + 1] = z
    
    elif method == "improved_euler":  
        for n in range(N):
            tn, yn = t[n], y[n]
            k1 = f(tn, yn)  
            k2 = f(tn+h, yn+h*k1) 
            y[n+1] = yn + h*(k1+k2)/2 



    return t, y  # 返回时间和解，便于后续误差对齐

# ==================== 3. 主程序：测试与验证 ====================
def main():
    # 【配置参数】步长列表
    h_list = [0.1, 0.05, 0.025, 0.0125, 0.00625]
    
    # 【方法选择】
    methods = ["trapezoid", "BDF2","improved_euler"]

    print("=" * 80)
    print("数值方法求解与误差验证报告")
    print("ODE: y' + e^(10t)y = e^(10t)sin(t) + cos(t), y(0)=0, t in (0, 1]")
    print("=" * 80)

    for method in methods:
        print(f"\n=== 方法：{method} ===")
        all_max_errors = []
        
        # 遍历步长列表
        for h in h_list:
            try:
                # 【调用主算法得到结果】
                t_num, y_num = Solve_Step(h, method)
                
                # 【计算绝对误差 (点态 + 最大)】
                # 精确解直接传入时间数组
                exact_vals = y_exact(t_num)
                abs_errors = np.abs(y_num - exact_vals)
                max_err = np.max(abs_errors)
                
                all_max_errors.append(max_err)
                
                print(f"步长 h={h:.5f} | Max Absolute Error: {max_err:.2e}")

            except Exception as e:
                print(f"步长 h={h:.5f} 发生异常：{e}")
        
        # 【计算收敛阶】(取相邻步长的误差比)
        if len(all_max_errors) >= 2:
            p_values = []
            for i in range(len(all_max_errors)-1):
                err_curr, err_prev = all_max_errors[i], all_max_errors[i+1]
                # h_current / h_previous = 0.5
                p = np.log(err_curr / err_prev) / np.log(2.0)
                p_values.append(p)
            
            print(f"收敛阶估计: [ {', '.join([f'{p:.4f}' for p in p_values])} ]")

        print("-" * 80)

    # 【最终比较分析】
    print("\n=== 结果比较与分析 ===")
    print("梯形法 (Trapezoid): A-稳定，二阶精度。对于刚性方程 e^(10t)，表现稳健，误差随 h² 衰减。")
    print("BDF2 格式：A-稳定，二阶精度。起步值需 RK4 提供，牛顿迭代求解隐式方程，计算量略大但精度相当。")
    print("观察：若梯形法收敛阶接近 2.0，BDF2 也接近 2.0，则说明两者在该问题下均发挥预期性能。")
    print("Improved Euler (显式): 条件稳定。")
    print("\n   - 刚性特征：由于 ODE 中包含项 -exp(10t)y，特征值 λ(t) ≈ -exp(10t)，在 t=1 时 λ ≈ -22026。")
    print("   - Improved Euler (显式) 的绝对稳定区域约为 h*λ > -4.5，即 h < 0.0002 * exp(-10t)。")
    print("   - 结论：在 t=1 附近，Improved Euler 需要极小的步长才能维持稳定，否则误差会随阶数平方指数增长。")
    print("=" * 80)

if __name__ == "__main__":
    main()
\end{lstlisting}
