17   def rk4(f, x0 , x_end , y0 , h):
18       """ 经 典 四 阶 Runge -- Kutta ； 标 量 与 NumPy 向 量 初 值 均 可 使 用 。 """
19       n_steps = round (( x_end - x0) / h)
20       if not np. isclose (x0 + n_steps * h, x_end ):
21           raise ValueError (" 步 长 h 必 须 整 除 积 分 区 间 长 度 。 ")
22

23       x = x0 + np. arange ( n_steps + 1) * h
24       y0_array = np. asarray (y0 , dtype=float)
25       y = np. zeros (( n_steps + 1,) + y0_array .shape , dtype= float )
26       y[0] = y0_array
27

28       for n in range( n_steps ):
29           k1 = f(x[n], y[n])
30           k2 = f(x[n] + h / 2, y[n] + h * k1 / 2)
31           k3 = f(x[n] + h / 2, y[n] + h * k2 / 2)
32           k4 = f(x[n] + h, y[n] + h * k3)
33           y[n + 1] = y[n] + h * (k1 + 2 * k2 + 2 * k3 + k4) / 6
34       return x, y
