4    def f(t, y):
5          """ODE 右 端 项 ： y' = -exp (10t)*(y-sin(t)) + cos(t)"""
6          return -np.exp (10 * t) * (y - np.sin(t)) + np.cos(t)

8    def fy(t, y):
9          """ 偏 导 数 � f / � y ： 用 于 牛 顿 迭 代 和 隐 式 方 程 求 解 """
